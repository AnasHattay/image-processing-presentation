FFT on Signals (2)

import numpy as np import matplotlib.pyplot as plt from matplotlib.colors import LinearSegmentedColormap # Parameters freq_x = 3 # Frequency in X direction freq_y = 2 # Frequency in Y direction amplitude = 1.0 # Wave amplitude signal_type = "X" # Signal type: "X" or "Y" # Create coordinate grids x = np.linspace(0, 2*np.pi, 128) y = np.linspace(0, 2*np.pi, 128) X, Y = np.meshgrid(x, y) # Generate 2D sinewave based on signal type if signal_type == "X": # Vertical stripes (horizontal frequency) sinewave = amplitude * np.sin(freq_x * X) elif signal_type == "Y": # Horizontal stripes (vertical frequency) sinewave = amplitude * np.sin(freq_y * Y) # Compute 2D FFT fft_result = np.fft.fft2(sinewave) fft_shifted = np.fft.fftshift(fft_result) magnitude_spectrum = np.abs(fft_shifted) # Display results fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 5)) ax1.imshow(sinewave, cmap='viridis') ax1.set_title(f'2D Sinewave ({signal_type})') ax2.imshow(np.log(magnitude_spectrum + 1), cmap='viridis') ax2.set_title('Magnitude Spectrum (log scale)') plt.show()
